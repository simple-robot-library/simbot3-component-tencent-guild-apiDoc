在 ApplicationBuilder.bots 作用域中寻找并使用 TencentGuildBotManager。

simpleApplication {
    bots {
       tencentGuilds {
           val bot = register("app id", "app key", "token") {
               // config
           }
           bot.start()
       }
    }
}
